14  public class PlayerBoardController extends BoardController {
15      private FleetInitializer fleetInitializer;
16  
17      public PlayerBoardController(GameBoard gameBoard, Game game) {
18          super(gameBoard, game);
19      }
20  
21      @Override
22      public void buildBoard() {
23          GameBoardBuilder gameBoardBuilder = new GameBoardBuilder(this);
24          fleetInitializer = new FleetInitializer(this);
25          gameBoardBuilder.fillTheBoardWithButtonsAndLabels();
26      }
27  
28      public void placeShips() {
29          Map<Integer, BoardTile> indexTiles = gameBoard.getIndexTiles();
30          Platform.runLater(() -> indexTiles.forEach((integer, boardTile) -> boardTile.setTileStyle(Styles.DEFAULT_TILE_COLOR, Styles.TEXT_BLACK)));
31          fleetInitializer.setUpShips();
32      }
33  
34      public String getFleetMessageForServer() {
35          return ServerMessageCreator.createFleetMessage(fleetInitializer.getShipPlaces());
36      }
37  }
