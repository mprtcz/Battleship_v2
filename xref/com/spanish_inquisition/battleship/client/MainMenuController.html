97      private void acceptANameAndStartNewGame(String text) {
98          setUpOnCloseRequest();
99          setUpSocketConnection();
100         game.acceptPlayersName(text);
101     }
102 
103     private void setUpSocketConnection() {
104         this.game = new Game();
105         game.setScoreListView(scoresListView);
106         game.setStatusController(new StatusController(playersLabel));
107         socketProgressIndicator.setVisible(true);
108         try {
109             gameStatusLabel.setText("Trying to connect to the server: " + hostIP);
110             featureButton.setText("Connecting...");
111             socketClient = SocketClient.createSocketClientWithSocket(hostIP, port);
112             game.setSocketClient(socketClient);
113             socketProgressIndicator.setVisible(false);
114             gameStatusLabel.setText("");
115             playerNameVBox.setVisible(false);
116             buildPlayerBoard();
117         } catch (IOException e) {
118             logger.log(Level.WARNING,
119                     "The client could not connect to the server", e);
120             gameStatusLabel.setText("I couldn't connect to the server");
121             socketProgressIndicator.setVisible(false);
122             featureButton.setText("Send");
123         }
124 
125     }
126 
127     private void setUpOnCloseRequest() {
128         Scene scene = mainBorderPane.getScene();
129         if (scene == null) {
130             return;
131         }
132         Window window = scene.getWindow();
133         window.setOnCloseRequest(event -> {
134             if (game != null) {
135                 game.stopGameRunning();
136                 game.closeSocketConnection();
137             }
138             Platform.exit();
139         });
140     }
141 
142     private void buildPlayerBoard() {
143         Platform.runLater(() -> playersLabel.setText("Set up your ships"));
144         game.buildPlayersBoard(new PlayerBoardController(
145                 new GameBoard(this.playersGridPane), game));
146         fleetSetupButton.setVisible(true);
147     }
148 
149 
150     @FXML
151     public void onFleetSetupButtonClicked() {
152         game.placePlayersShips();
153         sendToServerButton.setVisible(true);
154     }
155 
156     @FXML
157     void onSendToServerButtonClicked() {
158         fleetSetupButton.setVisible(false);
159         sendToServerButton.setVisible(false);
160         playersGridPane.setDisable(true);
161         opponentsGridPane.setDisable(true);
162         Platform.runLater(() -> playersLabel.setText("Wait for opponent's ships placement"));
163 
164         game.buildOpponentsBoard(new OpponentBoardController(
165                 new GameBoard(opponentsGridPane), game));
166         game.sendTheFleetToServer();
167         new Thread(() -> {
168             try {
169                 game.runTheGame();
170             } catch (InterruptedException e) {
171                 logger.log(DEFAULT_LEVEL, "Game interrupted");
172                 Thread.currentThread().interrupt();
173             }
174         }).start();
175     }
