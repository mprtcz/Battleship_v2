1   package com.spanish_inquisition.battleship.client;

3   import com.spanish_inquisition.battleship.common.AppLogger;
4   import javafx.application.Application;
5   import javafx.fxml.FXMLLoader;
6   import javafx.scene.Parent;
7   import javafx.scene.Scene;
8   import javafx.stage.Stage;
9   
10  import java.util.List;
11  
12  /**
13   * @author Michal_Partacz
14   * A main class for Client which will load the fxml file containing the UI's elements and shows them on the new window.
15   * The class also contains information about the window's height and width
16   */
17  public class Client extends Application {
18  
19      @Override
20      public void start(Stage primaryStage) throws Exception {
21          FXMLLoader loader = new FXMLLoader(getClass().getResource("/client/MainMenu.fxml"));
22          Parent root = loader.load();
23  
24          Scene scene = new Scene(root);
25          setUpPortAndIpInController(loader);
26          primaryStage.setTitle("Battleships");
27          primaryStage.setScene(scene);
28          primaryStage.show();
29      }
30  
31      public static void main(String[] args) {
32          AppLogger.initializeLogger();
33          launch(args);
34      }
35  
36      private void setUpPortAndIpInController(FXMLLoader loader) {
37          Parameters parameters = getParameters();
38          List<String> argsRaw = parameters.getRaw();
39          MainMenuController mainMenuController = loader.getController();
40          String serverIP = argsRaw.isEmpty() ? "localhost" : argsRaw.get(0);
41          mainMenuController.setHostIP(serverIP);
42          if (argsRaw.size() > 1) {
43              int port = Integer.parseInt(argsRaw.get(1));
44              mainMenuController.setPort(port);
45          }
46      }
47  }
