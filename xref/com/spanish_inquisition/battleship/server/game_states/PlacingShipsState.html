21  public class PlacingShipsState extends GameState {
22      public PlacingShipsState(Players players, MessageBus requestBus) {
23          super(players, requestBus);
24      }
25  
26      @Override
27      public GameState transform() {
28          List<Player> notReadyPlayers = new LinkedList<>(players.getBothPlayers());
29          while (!notReadyPlayers.isEmpty()) {
30              for (Player player : notReadyPlayers) {
31                  if (requestBus.haveMessageFromSender(player.getPlayerId())) {
32                      resolvePlayerFleetValidation(player, notReadyPlayers);
33                  }
34              }
35          }
36          sendPlayersScores();
37          return new PlayerActionState(players, requestBus);
38      }
39  
40      private void resolvePlayerFleetValidation(Player player, List<Player> notReadyPlayers) {
41          String fleetMessage = requestBus.getMessageFrom(player.getPlayerId()).getContent();
42          boolean isValidFleet = validateFleet(fleetMessage);
43          if (isValidFleet) {
44              player.setFleet(new FleetBuilder().build(fleetMessage));
45              notReadyPlayers.remove(player);
46              requestBus.addMessage(SERVER_ID, player.getPlayerId(), Header.FLEET_VALID.name());
47          } else {
48              requestBus.addMessage(SERVER_ID, player.getPlayerId(), Header.FLEET_INVALID.name());
49          }
50      }
51  
52      private boolean validateFleet(String fleetMessage) {
53          return fleetMessage.contains(Header.FLEET_REQUEST.name()) && FleetValidator.validate(fleetMessage);
54      }
55  
56      public void sendPlayersScores() {
57          PlayerEntityController playerEntityController = new PlayerEntityController(new DatabaseController());
58          List<PlayerEntity> playerRecords = playerEntityController.getPlayersFromDB();
59          logger.log(DEFAULT_LEVEL, "Read records from db:" + playerRecords);
60          for (PlayerEntity playerEntity: playerRecords) {
61              NetworkMessage networkMessage = new NetworkMessage(Header.SCORE, playerEntity.getReadyToSendRepresentation());
62              requestBus.addMessage(SERVER_ID, players.getCurrentPlayer().getPlayerId(), NetworkMessage.Parser.parseMessageToString(networkMessage));
63              requestBus.addMessage(SERVER_ID, players.getOpponentOf(players.getCurrentPlayer()).getPlayerId(),
64                      NetworkMessage.Parser.parseMessageToString(networkMessage));
65          }
66      }
67  }
