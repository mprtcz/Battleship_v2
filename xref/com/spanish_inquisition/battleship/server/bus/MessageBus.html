9   import java.util.Iterator;
10  import java.util.Optional;
11  import java.util.Queue;
12  import java.util.concurrent.ConcurrentLinkedQueue;
13  
14  public class MessageBus {
15      private Queue<Message> messageBus;
16  
17      public MessageBus() {
18          messageBus = new ConcurrentLinkedQueue<>();
19          AppLogger.initializeLogger();
20      }
21  
22      public void addMessage(Message message) {
23          messageBus.add(message);
24      }
25  
26      public void addMessage(int senderId, int recipientId, String message) {
27          Message newMessage = new Message(senderId, recipientId, message);
28          messageBus.add(newMessage);
29      }
30  
31      public Message getMessageFor(int recipientId) {
32          Message msg = null;
33          Iterator<Message> iterator = messageBus.iterator();
34          while (iterator.hasNext()) {
35              Message current = iterator.next();
36              if (current.isToRecipient(recipientId)) {
37                  msg = current;
38                  iterator.remove();
39                  break;
40              }
41          }
42          return msg;
43      }
44  
45      public Message getMessageFrom(int senderId) {
46          Optional<Message> optionalMessage;
47          optionalMessage = messageBus.stream()
48                  .filter(message -> message.isFromSender(senderId)).findFirst();
49          //removeAllSenderMessagesFromBus(senderId);
50          //messageBus.remove(optionalMessage);
51          Message message = optionalMessage.orElse(new Message(0, 0, ""));
52          removeMessage(message);
53          return message;
54      }
55  
56      private void removeMessage(Message forRemoveMessage) {
57          messageBus.removeIf(message -> message.equals(forRemoveMessage));
58      }
59  
60      private void removeAllSenderMessagesFromBus(int senderId) {
61          messageBus.removeIf(message -> message.isFromSender(senderId));
62      }
63  
64      public boolean haveMessageFromSender(int senderId) {
65          boolean isFromSender = false;
66          if (messageBus != null) {
67              isFromSender = messageBus.stream()
68                      .anyMatch(message -> message.isFromSender(senderId));
69          }
70          return isFromSender;
71      }
72  
73      public boolean haveMessageForRecipient(int recipientId) {
74          boolean isForRecipient = false;
75          if (messageBus != null) {
76              isForRecipient = messageBus.stream()
77                      .anyMatch(message -> message.isToRecipient(recipientId));
78          }
79          return isForRecipient;
80      }

82  }
