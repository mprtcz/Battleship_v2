13      static List<Integer> parseMessageToIntegersList(String message) {
14          String[] messageSplitted = message.split(String.valueOf(NetworkMessage.RESPONSE_HEADER_SPLIT_CHARACTER));
15          return parseStringArrayToIntegerList(StringToStringArrayParser.parse(messageSplitted[1]));
16      }
17  
18      private static List<Integer> parseStringArrayToIntegerList(String[] fleet) {
19          List<Integer> intList = new ArrayList<>(fleet.length);
20          for (String shipIndex : fleet) {
21              try {
22                  intList.add(Integer.parseInt(shipIndex));
23              } catch (NumberFormatException e) {
24                  logger.log(Level.WARNING, "wrong number format", e);
25              }
26          }
27          return intList;
28      }
