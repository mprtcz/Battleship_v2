7   /**
8    * @author Michal_Partacz
9    * A enum that will be sent with the requests and response to ther server, a server will identify the message by it's
10   * header value
11   */
12  public enum Header {
13      EXIT,
14      NAME,
15      UNKNOWN,
16      FLEET_REQUEST,
17      FLEET_INVALID,
18      FLEET_VALID,
19      PLAYER_TURN,
20      OPPONENT_TURN,
21      GAME_WON,
22      SCORE,
23      DECIDE_ON_MOVE,
24      MOVE_REGULAR,
25      RESPONSE_HIT,
26      RESPONSE_OPPONENT_HIT,
27      RESPONSE_MISS,
28      RESPONSE_OPPONENT_MISS,
29      RESPONSE_DESTROYED_SHIP,
30      RESPONSE_OPPONENT_DESTROYED_SHIP;
31  
32      /**
33       * A method which will return a Header enum for an input string. In case there is no enum that can be inferred from
34       * the string, it will return an UNKNOWN enum value.
35       *
36       * @param responseHeader a string which corresponds to the Headers name
37       * @return a Header enum value parsed from the string
38       */
39      public static Header parseResponseHeader(String responseHeader) {
40          Header header;
41          try {
42              header = valueOf(responseHeader);
43          } catch (IllegalArgumentException e) {
44              logger.log(Level.WARNING, e.getMessage());
45              header = UNKNOWN;
46          }
47          return header;
48      }
49  
50      public static boolean isResponseFieldChanging(Header header) {
51          // implement HIT / MISS / DESTROYED headers and return true if the header is one of those
52          return false;
53      }
